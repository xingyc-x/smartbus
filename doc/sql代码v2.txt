1、项目
CREATE TABLE project
(
id int NOT NULL,
name varchar(255),
palce varchar(255),
image_path varchar(255),
belong varchar(255),
introduction varchar(1000),
PRIMARY KEY (id)
)

2、楼层
CREATE TABLE floor
(
id varchar(255) NOT NULL,
palce varchar(255),
image_path varchar(255),
project_id bigint(20) NOT NULL,
PRIMARY KEY (id)
)

3、数据发送器
CREATE TABLE sender
(
mac varchar(255),
name varchar(255),
palce varchar(255),
floor_id varchar(255) NOT NULL,
PRIMARY KEY (mac)
)

4、数据点
CREATE TABLE data_point
(
id varchar(255) NOT NULL,
name varchar(255),
palce varchar(255),
unit varchar(255),
upper_bound float,
lower_bound float,
sender_mac varchar(255) NOT NULL,
PRIMARY KEY (id)
)

5、用户
CREATE TABLE user
(
name varchar(255) NOT NULL,
password varchar(255) NOT NULL,
tel varchar(255),
email varchar(255),
capacity varchar(255),
PRIMARY KEY (name)
)

6、操作记录
CREATE TABLE operation_log
(
id bigint(20) NOT NULL AUTO_INCREMENT,
content varchar(500),
user_name varchar(255),
PRIMARY KEY (id)
)

7、报表
CREATE TABLE report_form
(
id bigint(20) NOT NULL AUTO_INCREMENT,
content varchar(5000),
project_id bigint(20) NOT NULL,
PRIMARY KEY (id)
)

8、项目用户关系表
CREATE TABLE project_user_relation
(
id bigint(20) NOT NULL AUTO_INCREMENT,
project_id bigint(20),
user_name varchar(255),
PRIMARY KEY (id)
)
